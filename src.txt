let

    DataFim    = #date(2026, 12, 31),
    TotalDias  = Duration.Days(DataFim - DataInicio) + 1,
    ListaDatas = List.Dates(DataInicio, TotalDias, #duration(1,0,0,0)),
    Tabela     = Table.FromList(ListaDatas, Splitter.SplitByNothing(), {"date_full"}),
    TipoData   = Table.TransformColumnTypes(Tabela, {{"date_full", type date}}),

    AddDateSK  = Table.AddColumn(TipoData, "date_sk", each
                    Date.Year([date_full]) * 10000
                    + Date.Month([date_full]) * 100
                    + Date.Day([date_full]), Int32.Type),

    AddAno     = Table.AddColumn(AddDateSK,  "ano",           each Date.Year([date_full]),            Int32.Type),
    AddTrim    = Table.AddColumn(AddAno,     "trimestre",     each Date.QuarterOfYear([date_full]),   Int32.Type),
    AddTrimNm  = Table.AddColumn(AddTrim,    "trimestre_nome",each "Q" & Text.From(Date.QuarterOfYear([date_full])) & " " & Text.From(Date.Year([date_full])), type text),
    AddMes     = Table.AddColumn(AddTrimNm,  "mes",           each Date.Month([date_full]),           Int32.Type),
    AddMesNm   = Table.AddColumn(AddMes,     "mes_nome",      each Date.ToText([date_full], "MMMM", "pt-BR"), type text),
    AddSemana  = Table.AddColumn(AddMesNm,   "semana_ano",    each Date.WeekOfYear([date_full]),      Int32.Type),

    AddFimSem  = Table.AddColumn(AddDiaNm,   "flag_fds",      each if Date.DayOfWeek([date_full], Day.Monday) >= 5 then 1 else 0, Int32.Type),
    AddDia     = Table.AddColumn(AddFimSem,  "dia",           each Date.Day([date_full]),             Int32.Type)
