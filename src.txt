
'Total Pontos Previsto' =

SUMX(
    SUMMARIZE(
        fStory,
        Dim_Sprint[sprint_nk],
        "Status",   MAX( Dim_Sprint[sprint_state_desc] ),
        "Escopo",   SUM( fStory[story_points] ),
        "Comprom",  CALCULATE(
                        SUM( fStory[story_points] ),
                        fStory[flag_cancelada] = 0
                    )
    ),
    SWITCH(
        TRUE(),
        [Status] = "Complete", [Comprom],
        [Escopo]
    )
)
